# 安装 - Installation

# 下载

yarn add @wanmaoor/giaoui

# or

npm i @wanmaoor/giaoui

# 使用

在入口文件引入css样式:

  import "@wanmaoor/giaoui/dist/giaoui.css"

在组件里引用giaoui:

  import {Button} from "@wanmaoor/giaoui"
